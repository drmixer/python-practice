The Tally Pattern

‚ñº

Counting Occurrences

names = ["Alice", "Bob", "Alice", "Charlie", "Bob", "Alice"]

for name in names:
    counts[name] = counts.get(name, 0) + 1

print(counts)  # {'Alice': 3, 'Bob': 2, 'Charlie': 1}
üí° The Pattern: counts[key] = counts.get(key, 0) + 1
This gets the current count (or 0 if key doesn't exist) and adds 1.

Running Total (Score Tracker)

Step-by-Step Breakdown

# First iteration: name = "Alice"
	counts.get("Alice", 0)  # Returns 0 (not in dict yet)
	counts["Alice"] = 0 + 1  # counts = {"Alice": 1}

# Second iteration: name = "Bob"
counts.get("Bob", 0)    # Returns 0
counts["Bob"] = 0 + 1    # counts = {"Alice": 1, "Bob": 1}

# Third iteration: name = "Alice" (again!)
counts.get("Alice", 0)  # Returns 1 (already exists)
counts["Alice"] = 1 + 1  # counts = {"Alice": 2, "Bob": 1}

def collect_responses():
    """Collect survey responses until user types 'done'"""
    responses = []
    
    print("Enter responses (type 'done' to finish):")
    while True:
        response = input("Response: ").strip().lower()
        
        if response == "done":
            break
        
        if response == "":
            print("Please enter a response")
            continue
        
        responses.append(response)
    
    return responses

def tally_responses(responses):
    """Count occurrences of each response"""
    counts = {}
    for response in responses:
        counts[response] = counts.get(response, 0) + 1
    return counts

def find_most_common(counts):
    """Find the response with highest count"""
    if len(counts) == 0:
        return None, 0
    
    max_response = None
    max_count = 0
    
    for response, count in counts.items():
        if count > max_count:
            max_count = count
            max_response = response
    
    return max_response, max_count

def display_results(counts, total):
    """Display results with percentages"""
	    print("
=== Survey Results ===")
	    print(f"Total responses: {total}
")
    
    for response, count in counts.items():
        percentage = (count / total) * 100
        print(f"{response:<15} {count:>4} ({percentage:>5.1f}%)")
    
    most_common, count = find_most_common(counts)
	    print(f"
Most common: {most_common} ({count} votes)")

# Main program
print("=== Survey: Favorite Programming Language ===")
responses = collect_responses()

if len(responses) == 0:
    print("No responses collected.")
else:
    counts = tally_responses(responses)
    display_results(counts, len(responses))

def collect_temperatures():
    """Collect daily temperatures for a week"""
    temps = []
    days = ["Monday", "Tuesday", "Wednesday", "Thursday", 
            "Friday", "Saturday", "Sunday"]
    
    for day in days:
        while True:
            try:
                temp = float(input(f"{day} temperature (¬∞F): "))
                if temp < -50 or temp > 150:
                    print("Temperature seems unrealistic")
                    continue
                temps.append(temp)
                break
            except ValueError:
                print("Invalid input. Enter a number.")
    
    return temps

def analyze_temperatures(temps):
    """Analyze temperature data"""
    avg_temp = sum(temps) / len(temps)
    max_temp = max(temps)
    min_temp = min(temps)
    
    # Count days above/below average
    above_avg = 0
    below_avg = 0
    for temp in temps:
        if temp > avg_temp:
            above_avg += 1
        elif temp < avg_temp:
            below_avg += 1
    
    return {
        "average": avg_temp,
        "maximum": max_temp,
        "minimum": min_temp,
        "above_avg": above_avg,
        "below_avg": below_avg
    }

Example 6: Book Ratings Summary (List of Dicts + Input Validation)

Covers:functionslistsdictionariesfor loopwhile looptry/exceptinput validationf-stringsformattingsum/len

def calc_avg(books):
    # Sums up the "rating" for each book in the list of dicts,
    # then divides by the number of books to get the average.
    return sum(book["rating"] for book in books) / len(books)

def main():
    books = []  # Holds all book dicts

    # ---- INPUT: How many books? ----
    while True:  # Loop until user enters a valid number
        try:
            qty = int(input("How many books?: "))   # <- Must convert input to int!
            break
        except ValueError:
            print("Numbers only.")                  # <- Catches non-integer input

    # ---- Book data entry ----
    for _ in range(qty):
        title = input("Enter a title: ")            # <- "title" (no quotes) is variable, not dict key!

        # INPUT: Rating with validation (must be int between 1 and 5)
        while True:  # Loop until valid input
            try:
                rating = int(input("Enter a rating 1-5: "))
                if 1 <= rating <= 5:
                    break  # Good rating, move on
                else:
                    print("Rating must be 1-5.")    # <- Needs to be in valid range
            except ValueError:
                print("Numbers only.")               # <- Handles non-integer input

        # ---- Building a Book Dict ----
        book = {"title": title, "rating": rating}    # <- Key "title" (in quotes), value is variable title (no quotes)
        books.append(book)                           # <- Add book dict to "books" list

    # ---- Printing Section Header with Alignment ----
    print("\n" + "-" * 28)
    print(f"{'SUMMARY':^28}")  # <- Centers the word "SUMMARY" in 28 spaces
    print("-" * 28)
    print(f"{'Title':<20} {'Rating':>7}")  # <- Left-aligned title, right-aligned rating header
    print("-" * 28)

    # ---- Printing Each Book ----
    for book in books:
        print(f"{book['title']:<20} {book['rating']:>7}")  # <- "book['title']" is the key lookup in the dict

    # ---- Average Calculation ----
    avg = calc_avg(books)  # Grabs the average rating from the helper

    # ---- Printing Average Rating with Two Decimals ----
    print(f"\n{'*' * 5} AVG RATING: {avg:.2f} {'*' * 5}")  # <- ":.2f" keeps two decimals, a classic pattern for money/averages



if __name__ == "__main__":
    main()

‚ö° Key Reminders for the Exam

‚ñº
1. input() always returns a string
You MUST convert to int or float: age = int(input("Age: "))
2. range(5) is 0, 1, 2, 3, 4 (NOT 5!)
The stop value is exclusive. For 1-5 use: range(1, 6)
3. Use .get(key, default) for safe dictionary access
count = counts.get(name, 0) prevents KeyError
4. Don't forget 'f' in f-strings!
Correct: f"Hello {name}"
Wrong: "Hello {name}" (prints literally)
5. Return vs Print in functions
return - sends value back to caller (use for calculations)
print - displays to console (use for output only)
6. Tally pattern formula
counts[key] = counts.get(key, 0) + 1
7. While loop with break
Always use: while True: + if condition: break
8. Checking even/odd
if num % 2 == 0: (even)
else: (odd)
9. List indexing starts at 0
First element: mylist[0]
Last element: mylist[-1]
10. Always check for empty lists before averaging
if len(scores) > 0: before dividing
11. String alignment in f-strings
:<10 left-align (10 spaces)
:>10 right-align (10 spaces)
:.2f two decimal places
12. When tracing code, work line by line
Write down variable values after each line to avoid mistakes
13. Division operators
/ gives float (10 / 3 = 3.333...)
// gives int (10 // 3 = 3)
14. Common mistakes to avoid
‚Ä¢ Forgetting to convert input() to int/float
‚Ä¢ Using = instead of == in conditions
‚Ä¢ Forgetting to update loop counter in while loops
‚Ä¢ Not handling empty lists before calculating average
‚Ä¢ Off-by-one errors with range() (remember: exclusive!)
